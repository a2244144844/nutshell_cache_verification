## 1. DIR-014: Probe Hit Full Release

### 1.1 RTL 行为分析

```
CacheStage3 (Cache.v)
├── state == s_idle (0x0)
│   ├── if (probe)                          # line 767: io_in_bits_req_cmd == PROBE(0x8)
│   │   ├── if (_T_27)                      # line 768: io_out_coh_resp fire (ready & valid)
│   │   │   ├── state <= _state_T           # line 769: hit ? s_release(0x8) : s_idle(0x0)
│   │   │   └── readBeatCnt <= addr_wordIdx # line 797: init release beat counter
│   │   └── ...
│   └── ...
├── state == s_release (0x8)
│   ├── io_out_coh_resp_valid = 1           # sends release data beats
│   ├── releaseLast_c_value increments      # line 865: each cohResp.fire
│   ├── releaseLast = (counter==7) & fire   # line 602: last beat indicator
│   └── resp cmd: 0xC=hit data, 0x6=last   # line 603-604
└── releaseLast counter: lines 598-602
```

**关键信号**:
- `probe` = `io_in_valid & (req_cmd == 0x8)` — line 511
- `_T_27` = `io_out_coh_resp_ready & io_out_coh_resp_valid` — cohResp handshake
- `_state_T` = `hit ? 0x8 : 0x0` — line 611
- `releaseLast` = `_releaseLast_T_2 & (counter == 7)` — line 602

### 1.2 现有测试的问题

`test_probe_hit_returns_correct_data` 在收到第一个 `coh_resp_valid` 后就 break 了，**从未进入 `s_release` 状态**。

### 1.3 实现方案

在 `tests/directed/test_coherence_probe.py` 中新增测试函数：

```python
@toffee_test.testcase
async def test_probe_hit_full_release_sequence(cache_env):
    """DIR-014: Probe hit → full 8-beat release sequence.
    Covers: 767-769, 795-797, 598-602, 865 (s_release counter)."""
    env = cache_env
    env.reset()

    addr = 0x8000_1000
    fill_data = 0xDEAD_BEEF_CAFE_BABE

    # Step 1: Fill a cache line (ensures probe will hit)
    resp, mem = env.send_cpu_request(
        sb.CpuRequest(cmd=sb.READ, addr=addr, user=0x701),
        refill_data=fill_data,
    )
    env.scoreboard.check_read_response(resp, expected_data=fill_data, expected_user=0x701)
    env.scoreboard.check_single_read_burst(mem, expected_addr=addr)

    # Step 2: Set coh_resp_ready BEFORE driving probe
    env.set_pin("io_out_coh_resp_ready", 1)

    # Step 3: Drive probe request
    env.set_pin("io_out_coh_req_bits_addr", addr)
    env.set_pin("io_out_coh_req_bits_size", 3)
    env.set_pin("io_out_coh_req_bits_cmd", sb.PROBE)
    env.set_pin("io_out_coh_req_bits_wmask", 0)
    env.set_pin("io_out_coh_req_bits_wdata", 0)
    env.set_pin("io_out_coh_req_valid", 1)

    # Step 4: Wait for probe accepted, then count ALL coh_resp beats
    accepted = False
    beat_count = 0
    last_valid = False
    idle_cycles = 0
    for _ in range(300):
        will_accept = bool(
            env.get_pin("io_out_coh_req_valid")
            and env.get_pin("io_out_coh_req_ready")
        )
        cur_valid = bool(env.get_pin("io_out_coh_resp_valid"))

        # Count rising edges of coh_resp_valid
        if cur_valid and not last_valid:
            beat_count += 1
            idle_cycles = 0
        if not cur_valid and beat_count > 0:
            idle_cycles += 1

        # Exit after release complete (valid low for enough cycles)
        if not cur_valid and idle_cycles > 15 and beat_count > 1:
            break

        last_valid = cur_valid
        env.step(1)

        if will_accept and not accepted:
            accepted = True
            env.set_pin("io_out_coh_req_valid", 0)

    assert accepted, "probe request was not accepted"
    # Expect: 1 probe response (cmd=0xC) + 8 release data beats = 9 total
    # But the exact count depends on how the DUT sequences them.
    # At minimum: probe response + at least 1 release beat.
    assert beat_count >= 2, (
        f"expected probe response + release beats (>=2), got {beat_count}"
    )
```

## 2. DIR-015: Read-Burst Hit

### 2.1 RTL 行为分析

```
CacheStage3 (Cache.v)
├── state == s_idle (0x0)
│   ├── hitReadBurst = hit & (cmd == READ_BURST)  # line 513
│   ├── if (!probe && _T_29)                       # line 771: hitReadBurst & io_out_ready
│   │   └── state <= s_release (0x8)               # line 772
│   └── readBeatCnt <= addr_wordIndex + 1 wrap     # line 800: critical-word-first
├── state == s_release (0x8)
│   ├── respToL1Fire = hitReadBurst & io_out_ready # line 605
│   ├── respToL1Last counter: lines 608-610        # 8-beat response counter
│   └── respToL1Last_c_value increment: line 870
└── io_in_resp sends 8 beats of cached data
```

**关键区别**: 不同于 probe release（通过 `io_out_coh_resp_*`），read-burst hit 通过 `io_in_resp_*` 返回数据。

### 2.2 为什么未覆盖

所有现有读测试使用 `cmd=READ(0x0)`，从未发送 `cmd=READ_BURST(0x2)` 到已缓存地址。

### 2.3 实现方案

新建 `tests/directed/test_read_burst_hit.py`：

```python
import sys
from pathlib import Path

ROOT_DIR = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(ROOT_DIR))

import toffee_test

from src.utils import simplebus as sb


@toffee_test.testcase
async def test_read_burst_hit_returns_8_beats(cache_env):
    """DIR-015: READ_BURST to a cached line triggers hit burst response.
    Covers: 513 (hitReadBurst), 605 (respToL1Fire), 608-610 (respToL1Last),
    771-772 (state <= s_release), 800 (readBeatCnt), 870 (counter inc)."""
    env = cache_env
    env.reset()

    addr = 0x8000_1000
    fill_data = 0xDEAD_BEEF_CAFE_BABE

    # Step 1: Fill a cache line with a normal READ
    resp, mem = env.send_cpu_request(
        sb.CpuRequest(cmd=sb.READ, addr=addr, user=0x501),
        refill_data=fill_data,
    )
    env.scoreboard.check_read_response(resp, expected_data=fill_data, expected_user=0x501)
    env.scoreboard.check_single_read_burst(mem, expected_addr=addr)

    # Step 2: Send READ_BURST to the same cached address
    # Use low-level pin driving since send_cpu_request may not support READ_BURST
    env.set_pin("io_in_req_bits_addr", addr)
    env.set_pin("io_in_req_bits_size", 3)
    env.set_pin("io_in_req_bits_cmd", sb.READ_BURST)  # 0x2
    env.set_pin("io_in_req_bits_wmask", 0)
    env.set_pin("io_in_req_bits_wdata", 0)
    env.set_pin("io_in_req_bits_user", 0x502)
    env.set_pin("io_in_req_valid", 1)

    # Step 3: Keep resp_ready high, count response beats
    env.set_pin("io_in_resp_ready", 1)

    accepted = False
    beat_count = 0
    last_valid = False
    idle_cycles = 0
    resp_cmds = []
    resp_rdata = []

    for _ in range(200):
        will_accept = bool(
            env.get_pin("io_in_req_valid")
            and env.get_pin("io_in_req_ready")
        )
        cur_valid = bool(env.get_pin("io_in_resp_valid"))

        if cur_valid and not last_valid:
            beat_count += 1
            resp_cmds.append(env.get_pin("io_in_resp_bits_cmd"))
            resp_rdata.append(env.get_pin("io_in_resp_bits_rdata"))
            idle_cycles = 0
        if not cur_valid and beat_count > 0:
            idle_cycles += 1

        if not cur_valid and idle_cycles > 15 and beat_count > 0:
            break

        last_valid = cur_valid
        env.step(1)

        if will_accept and not accepted:
            accepted = True
            env.set_pin("io_in_req_valid", 0)

    assert accepted, "READ_BURST request was not accepted"
    # Should get 8 response beats for a full cache line
    assert beat_count >= 8, (
        f"expected 8 response beats for READ_BURST hit, got {beat_count}"
    )
```

## 3. DIR-016: needFlush De-assertion

### 3.1 RTL 行为分析

```verilog
// line 558: reg needFlush;
// line 559: _GEN_1 = io_flush & state != 0 | needFlush;
// line 786-788:
//   if (reset) needFlush <= 0;
//   else if (_T_5 & needFlush) needFlush <= 0;  // _T_5 = io_out_ready & io_out_valid
//   else needFlush <= _GEN_1;
```

`needFlush` 是 sticky flag：
- **置位条件**: `io_flush & state != 0`（flush 到达时 CacheStage3 不在 idle）
- **清零条件**: `io_out_ready & io_out_valid & needFlush`（下一个请求完成握手时）

### 3.2 现有测试的问题

`test_flush_during_miss` 置位了 `needFlush`，但之后用 `send_cpu_request` 发新请求时，`send_cpu_request` 内部的 memory response 模拟可能在 `needFlush` 被观测到之前就完成了握手。需要确保新请求的 `io_out_valid & io_out_ready` 握手发生在 `needFlush` 仍为高的时刻。

### 3.3 实现方案

在 `tests/directed/test_flush_behavior.py` 中新增：

```python
@toffee_test.testcase
async def test_flush_during_miss_then_recover_with_subsequent_request(cache_env):
    """DIR-016: Flush during miss sets needFlush; new request clears it.
    Covers: 558 (needFlush reg), 788 (needFlush <= 0)."""
    env = cache_env
    env.reset()

    addr_a = 0x8000_2000
    addr_b = 0x8000_5000

    # Step 1: Start a READ miss (state leaves idle)
    env.drive_cpu_request(sb.CpuRequest(cmd=sb.READ, addr=addr_a, user=0x401))

    # Step 2: Assert flush while miss is in-flight → sets needFlush
    env.set_pin("io_flush", 0b01)

    # Step 3: Wait for request accepted
    accepted = False
    for _ in range(100):
        env.step(1)
        if env.get_pin("io_in_req_valid") and env.get_pin("io_in_req_ready"):
            accepted = True
            break
    assert accepted, "read miss request was not accepted"

    env.clear_cpu_request()

    # Step 4: Wait for pipeline to drain
    for _ in range(50):
        if env.get_pin("io_empty") == 1:
            break
        env.step(1)
    assert env.get_pin("io_empty") == 1, "io_empty should be high after flush"

    # Step 5: Deassert flush
    env.set_pin("io_flush", 0)
    env.step(10)

    # Step 6: Issue a NEW READ request to a different address
    # This request's completion handshake (io_out_ready & io_out_valid)
    # should trigger needFlush <= 0 at line 788
    resp, mem = env.send_cpu_request(
        sb.CpuRequest(cmd=sb.READ, addr=addr_b, user=0x402),
        refill_data=0xFEED_FACE_CAFE_BEEF,
    )
    env.scoreboard.check_read_response(resp, expected_data=0xFEED_FACE_CAFE_BEEF, expected_user=0x402)
    env.scoreboard.check_single_read_burst(mem, expected_addr=addr_b)
```

## 7. 注意事项

1. **不要修改 `src/` 下任何文件**。所有测试通过 `CacheEnv` 的公开 API 驱动。
2. **`send_cpu_request` 内部会自动处理 memory response**。对于 READ_BURST hit（DIR-015），因为是 hit 路径，不会产生 memory request，所以需要用低级 pin 驱动。
3. **`io_out_coh_resp_ready` 必须在驱动 probe 之前设为 1**。否则 DUT 的 release 数据会被 ready=0 阻塞。
4. **`io_flush` 只用 bit 0**（`0b01`）。bit 1 在 I-cache 配置下被断言阻塞。
5. **步进周期数要充足**。release 是 8 拍数据，加上 pipeline 延迟，循环上限建议 200-300。
